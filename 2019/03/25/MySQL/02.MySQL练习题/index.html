--1  查询" 01 "课程比" 02 "课程成绩高的学生的信息及课程分数
SELECT * FROM (
SELECT t1.SId, c1, c2 FROM (
SELECT SId ,score AS c1 FROM SC WHERE `CId` = 01) AS t1,(SELECT SId,score AS c2 FROM SC WHERE CId = 02) AS t2 WHERE t1.SId =t2.SId AND t1.c1 > t2.c2) r 
LEFT JOIN  Student ON Student.`SId` = r.SId;

--1.1  查询同时存在" 01 "课程和" 02 "课程的情况
SELECT * FROM(
(SELECT SId ,score AS c1 FROM SC WHERE `CId` = 01) AS t1,(SELECT SId,score AS c2 FROM SC WHERE CId = 02) AS t2)
WHERE t1.SId = t2.SId

--1.2  查询存在" 01 "课程但可能不存在" 02 "课程的情况(不存在时显示为 null )
SELECT * FROM
(SELECT SId ,score AS c1 FROM SC WHERE `CId` = 01) AS t1 LEFT JOIN (SELECT SId,score AS c2 FROM SC WHERE CId = 02) AS t2 ON t1.SId = t2.SId

--1.3 查询不存在" 01 "课程但存在" 02 "课程的情况
SELECT * FROM SC WHERE SC.SId NOT IN (SELECT SId FROM SC WHERE SC.CId = 01) AND SC.CId= 02;
